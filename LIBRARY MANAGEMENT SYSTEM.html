02_books_table.sql
-- ────────────────────────────────────────────────
-- TABLE: books  (Master catalog of all books)
-- ────────────────────────────────────────────────
CREATE TABLE books (
  book_id       INT AUTO_INCREMENT PRIMARY KEY,
  isbn          VARCHAR(20)  UNIQUE NOT NULL,
  title         VARCHAR(200) NOT NULL,
  author        VARCHAR(150) NOT NULL,
  publisher     VARCHAR(150),
  genre         VARCHAR(80),
  edition       VARCHAR(20),
  year_published YEAR,
  total_copies  INT          NOT NULL DEFAULT 1,
  available_copies INT       NOT NULL DEFAULT 1,
  shelf_location VARCHAR(50),
  description   TEXT,
  cover_url     VARCHAR(500),
  added_on      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
  updated_on    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP
                              ON UPDATE CURRENT_TIMESTAMP,
  CONSTRAINT chk_copies CHECK (available_copies <= total_copies
                              AND available_copies >= 0)
);

-- Index for fast title/author search
CREATE INDEX idx_books_title  ON books (title);
CREATE INDEX idx_books_author ON books (author);
CREATE INDEX idx_books_genre  ON books (genre);
03_members_table.sql
-- ────────────────────────────────────────────────
-- TABLE: members  (Registered library members)
-- ────────────────────────────────────────────────
CREATE TABLE members (
  member_id     INT  AUTO_INCREMENT PRIMARY KEY,
  member_code   VARCHAR(20)  UNIQUE NOT NULL,
  full_name     VARCHAR(150) NOT NULL,
  email         VARCHAR(120) UNIQUE NOT NULL,
  phone         VARCHAR(20),
  address       TEXT,
  gender        ENUM('Male', 'Female', 'Other'),
  date_of_birth DATE,
  membership_type ENUM('Student', 'Faculty', 'Public')
                  NOT NULL DEFAULT 'Student',
  is_active     TINYINT(1)   NOT NULL DEFAULT 1,
  max_books     INT          NOT NULL DEFAULT 3,
  joined_on     DATE         NOT NULL DEFAULT (CURDATE()),
  expiry_date   DATE,
  notes         TEXT,
  created_at    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_members_name  ON members (full_name);
CREATE INDEX idx_members_email ON members (email);
04_transactions_table.sql
-- ────────────────────────────────────────────────
-- TABLE: transactions  (Issue / Return log)
-- ────────────────────────────────────────────────
CREATE TABLE transactions (
  trans_id      INT  AUTO_INCREMENT PRIMARY KEY,
  book_id       INT  NOT NULL,
  member_id     INT  NOT NULL,
  issue_date    DATE NOT NULL DEFAULT (CURDATE()),
  due_date      DATE NOT NULL,
  return_date   DATE,
  fine_amount   DECIMAL(8,2) DEFAULT 0.00,
  fine_paid     TINYINT(1)   DEFAULT 0,
  status        ENUM('Issued', 'Returned', 'Overdue', 'Lost')
                NOT NULL DEFAULT 'Issued',
  remarks       TEXT,
  issued_by     VARCHAR(80)  -- Librarian name / staff ID,
  created_at    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP
                              ON UPDATE CURRENT_TIMESTAMP,

  FOREIGN KEY (book_id)   REFERENCES books(book_id)   ON DELETE RESTRICT,
  FOREIGN KEY (member_id) REFERENCES members(member_id) ON DELETE RESTRICT
);

CREATE INDEX idx_tx_book   ON transactions (book_id);
CREATE INDEX idx_tx_member ON transactions (member_id);
CREATE INDEX idx_tx_status ON transactions (status);
05_triggers.sql
-- ────────────────────────────────────────────────
-- TRIGGERS: Auto-update available_copies
-- ────────────────────────────────────────────────

-- Decrease available copies when book is issued
DELIMITER $$
CREATE TRIGGER trg_after_issue
AFTER INSERT ON transactions
FOR EACH ROW
BEGIN
  UPDATE books
  SET available_copies = available_copies - 1
  WHERE book_id = NEW.book_id;
END$$
DELIMITER ;

-- Increase available copies when book is returned
DELIMITER $$
CREATE TRIGGER trg_after_return
AFTER UPDATE ON transactions
FOR EACH ROW
BEGIN
  IF NEW.status = 'Returned' AND OLD.status != 'Returned' THEN
    UPDATE books
    SET available_copies = available_copies + 1
    WHERE book_id = NEW.book_id;
  END IF;

  -- Auto-calculate fine (₹2/day after due date)
  IF NEW.return_date > NEW.due_date THEN
    UPDATE transactions
    SET fine_amount = DATEDIFF(NEW.return_date, NEW.due_date) * 2.00
    WHERE trans_id = NEW.trans_id;
  END IF;
END$$
DELIMITER ;
06_views_and_queries.sql
-- ════════════════════════════════════════════════
-- USEFUL VIEWS & QUERIES
-- ════════════════════════════════════════════════

-- VIEW: currently issued books with member info
CREATE VIEW vw_issued_books AS
SELECT
  t.trans_id,
  b.title        AS book_title,
  b.author,
  m.full_name    AS member_name,
  m.email,
  t.issue_date,
  t.due_date,
  DATEDIFF(CURDATE(), t.due_date) AS days_overdue,
  t.status
FROM transactions t
JOIN books   b ON b.book_id   = t.book_id
JOIN members m ON m.member_id = t.member_id
WHERE t.status IN ('Issued', 'Overdue');

-- QUERY: overdue books with fines
SELECT
  m.full_name, m.phone, b.title,
  t.due_date,
  DATEDIFF(CURDATE(), t.due_date) AS days_late,
  DATEDIFF(CURDATE(), t.due_date) * 2 AS fine_due
FROM transactions t
JOIN books   b ON b.book_id   = t.book_id
JOIN members m ON m.member_id = t.member_id
WHERE t.status = 'Issued'
  AND t.due_date < CURDATE()
ORDER BY days_late DESC;

-- QUERY: most borrowed books
SELECT
  b.title, b.author,
  COUNT(t.trans_id) AS borrow_count
FROM transactions t
JOIN books b ON b.book_id = t.book_id
GROUP BY b.book_id
ORDER BY borrow_count DESC
LIMIT 10;

-- QUERY: members with most borrows
SELECT
  m.member_code, m.full_name,
  COUNT(t.trans_id) AS total_borrows
FROM transactions t
JOIN members m ON m.member_id = t.member_id
GROUP BY m.member_id
ORDER BY total_borrows DESC
LIMIT 5;
